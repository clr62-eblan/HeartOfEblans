decrease_corruption_law = {

	if = {
		limit = { has_idea = mafia_country }
		add_ideas = high_corruption
	}

	if = {
		limit = { has_idea = high_corruption }
		add_ideas = medium_corruption
	}

	if = {
		limit = { has_idea = medium_corruption }
		add_ideas = low_corruption
	}

	if = {
		limit = { has_idea = low_corruption }
		add_ideas = absolute_transparency
	}

}

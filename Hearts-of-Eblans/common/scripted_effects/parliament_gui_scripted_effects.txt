low_parliament_popularity_increase = {
	if = {
		limit = {
			check_variable = { deputatov_za < 105 }
		}
	
		random_list = {
			55 = {
				if = {
					limit = {
						check_variable = { deputatov_hz > 1 } # seats to be subtracted is less than 0 currently
					}
					add_to_variable = { deputatov_za = 5 }
					add_to_variable = { deputatov_protiv = -3}
					add_to_variable = { deputatov_hz = -2}
				}

					add_to_variable = { deputatov_za = deputatov_protiv }

			}

				}
				else_if = {
					limit = {
						check_variable = { deputatov_protiv > 8 } # seats to be subtracted is less than 0 currently
					}
					add_to_variable = { deputatov_za = 9 }

				}
				else = {
					add_to_variable = { deputatov_za = deputatov_protiv }

				}
			}
			15 = {
				if = {
					limit = {
						check_variable = { deputatov_hz > 4 } # seats to be subtracted is less than 0 currently
					}
					add_to_variable = { deputatov_za = 13 }
					add_to_variable = { deputatov_protiv = -8}
					add_to_variable = { deputatov_hz = -5}
				}
				else_if = {
					limit = {
						check_variable = { deputatov_protiv > 12 } # seats to be subtracted is less than 0 currently
